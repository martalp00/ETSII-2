operator load(C,P,A)
pre: cargo(C), plane(P), airport(A), at(C,A), at(P,A)
post: ~at(C,A), in(C,P)
operator unload(C,P,A)
pre: cargo(C), plane(P), airport(A),in(C,P), at(P,A)
post: at(C,A), ~in(C,P)
operator fly(P,From,To)
pre: plane(P), airport(From), airport(To), at(P,From)
post: ~at(P,From), at(P,To)
